Advanced Scala Chapters: Streams

Example #1: [0, 1, 0, 1, ...]

object Demo extends App {
  def sieve(s: Stream[Int]): Stream[Int] =
    s.head #:: sieve (s.tail filter (_ % s.head != 0))
  sieve (Stream.from(2)) take 10 foreach println
}
